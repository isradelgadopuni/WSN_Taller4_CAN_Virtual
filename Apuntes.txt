################################################################################################################

# Crear imagen

docker build -t wsn-taller-4-canv-img:parte1 .

################################################################################################################

# Levantar contenedor 

docker run -it \
  --name wsn_cont_canv_ubu_t4 \
  --hostname host_taller4_canv \
  --network host \
  --cap-add=NET_ADMIN \
  -e DISPLAY=$DISPLAY \
  -v /tmp/.X11-unix:/tmp/.X11-unix:rw \
  -v "$HOME/Documentos/WSN/Taller_4_CAN_Virtual/pcaps:/root/capturas:rw" \
  wsn-taller-4-canv-img:parte1

			üîç Desglose r√°pido:
			-it ‚Üí modo interactivo con terminal (para que puedas usar bash dentro).
			--name wsn_cont_canv_ubu_t4 ‚Üí nombre fijo del contenedor (puedes reusarlo luego).
			--network host ‚Üí comparte la red y permite crear vcan0 visible globalmente.
			--cap-add=NET_ADMIN ‚Üí le da permisos para manipular interfaces CAN.
			-e DISPLAY=$DISPLAY ‚Üí env√≠a tu variable de display al contenedor.
			-v /tmp/.X11-unix:/tmp/.X11-unix:rw ‚Üí monta el socket X11 (para mostrar ventanas GUI).
			wsn-taller-4-canv-img:parte1 ‚Üí la imagen que construiste.

################################################################################################################

# Reabrir contenedor

docker start -ai wsn_cont_canv_ubu_t4  

# Nuevas terminales

docker exec -it wsn_cont_canv_ubu_t4 bash
